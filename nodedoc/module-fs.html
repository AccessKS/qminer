Module: fs

File-system module.
Source:

Example
// import module
var fs = require('qminer').fs;
// open file in write mode
var fout = fs.openWrite('file.txt');
// write sync and close
fout.writeLine('example text');
fout.close();
// open file in read mode
var fin = fs.openRead('file.txt');
// read a line
var str = fin.readLine();
